public class Main {
    public static void main(String[] args) {
        int a = 10;
        int b = 5;
        boolean x = true;
        boolean y = false;

        System.out.println(a > b);
        System.out.println(a < b);
        System.out.println(a != b);
        System.out.println(a >= b);
        System.out.println(x && y);
        System.out.println(x || y);
        System.out.println(!x);
       
    }
}
